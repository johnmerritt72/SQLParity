**Goal:** Replace the raw DDL detail view for tables with a structured two-pane tree showing columns/indexes/constraints/triggers with diff highlighting; populate the server picker from SSMS registered servers; auto-populate a database dropdown after connecting.

Create `src/SQLParity.Vsix/Helpers/RegisteredServerHelper.cs`:

```csharp
using System;
using System.Collections.Generic;

namespace SQLParity.Vsix.Helpers
{
    /// <summary>
    /// Reads the list of registered servers from SSMS. Falls back to an empty
    /// list if the API is unavailable (e.g., running outside SSMS).
    /// </summary>
    public static class RegisteredServerHelper
    {
        public static List<string> GetRegisteredServerNames()
        {
            var servers = new List<string>();
            try
            {
                // The RegisteredServersStore reads from SSMS's internal storage
                var store = new Microsoft.SqlServer.Management.RegisteredServers.RegisteredServersStore(
                    new Microsoft.SqlServer.Management.Common.ServerConnection());

                var dbEngineGroup = store.DatabaseEngineServerGroup;
                CollectServers(dbEngineGroup, servers);
            }
            catch
            {
                // If the API fails (wrong context, permissions, etc.), return empty.
                // The user can still type a server name manually.
            }
            return servers;
        }

        private static void CollectServers(
            Microsoft.SqlServer.Management.RegisteredServers.ServerGroup group,
            List<string> servers)
        {
            if (group == null) return;

            foreach (Microsoft.SqlServer.Management.RegisteredServers.RegisteredServer rs in group.RegisteredServers)
            {
                if (!string.IsNullOrWhiteSpace(rs.ServerName) && !servers.Contains(rs.ServerName))
                    servers.Add(rs.ServerName);
            }

            foreach (Microsoft.SqlServer.Management.RegisteredServers.ServerGroup subGroup in group.ServerGroups)
            {
                CollectServers(subGroup, servers);
            }
        }
    }
}
```

**Note:** The `RegisteredServersStore` constructor may need a different overload or no arguments depending on the actual API in the SSMS 22 version. If the constructor fails, the implementer should try:
- `new RegisteredServersStore()` (no args)
- `RegisteredServersStore.LocalFileStore` (static property)
- Direct XML parsing of `RegSrvr.xml` as a fallback

The try/catch ensures the app doesn't crash if the API isn't available.

```csharp
using System;
using System.Collections.ObjectModel;
using System.Data.SqlClient;
using System.Threading.Tasks;
using System.Windows;
using SQLParity.Core.Model;
using SQLParity.Core.Project;
using SQLParity.Vsix.Helpers;

namespace SQLParity.Vsix.ViewModels
{
    public class ConnectionSideViewModel : ViewModelBase
    {
        private string _serverName = string.Empty;
        private string _databaseName = string.Empty;
        private string _label = string.Empty;
        private EnvironmentTag _tag = EnvironmentTag.Untagged;
        private bool _useWindowsAuth = true;
        private string _sqlLogin = string.Empty;
        private string _sqlPassword = string.Empty;
        private bool _isConnecting;
        private string _connectionError = string.Empty;

        public ConnectionSideViewModel()
        {
            ConnectCommand = new RelayCommand(_ => ConnectAsync(), _ => !IsConnecting && !string.IsNullOrWhiteSpace(ServerName));

            // Load registered servers
            try
            {
                var registered = RegisteredServerHelper.GetRegisteredServerNames();
                foreach (var s in registered)
                    AvailableServers.Add(s);
            }
            catch { }
        }

        public ObservableCollection<string> AvailableServers { get; } = new ObservableCollection<string>();
        public ObservableCollection<string> AvailableDatabases { get; } = new ObservableCollection<string>();
        public RelayCommand ConnectCommand { get; }

        public string ServerName
        {
            get => _serverName;
            set
            {
                if (SetProperty(ref _serverName, value))
                {
                    OnPropertyChanged(nameof(IsComplete));
                    AvailableDatabases.Clear();
                    ConnectionError = string.Empty;
                }
            }
        }

        public string DatabaseName
        {
            get => _databaseName;
            set
            {
                if (SetProperty(ref _databaseName, value))
                    OnPropertyChanged(nameof(IsComplete));
            }
        }

        public string Label
        {
            get => _label;
            set
            {
                if (SetProperty(ref _label, value))
                {
                    Tag = EnvironmentTagStore.SuggestTagFromLabel(value);
                    OnPropertyChanged(nameof(IsComplete));
                }
            }
        }

        public EnvironmentTag Tag
        {
            get => _tag;
            set => SetProperty(ref _tag, value);
        }

        public bool UseWindowsAuth
        {
            get => _useWindowsAuth;
            set => SetProperty(ref _useWindowsAuth, value);
        }

        public string SqlLogin
        {
            get => _sqlLogin;
            set => SetProperty(ref _sqlLogin, value);
        }

        public string SqlPassword
        {
            get => _sqlPassword;
            set => SetProperty(ref _sqlPassword, value);
        }

        public bool IsConnecting
        {
            get => _isConnecting;
            set => SetProperty(ref _isConnecting, value);
        }

        public string ConnectionError
        {
            get => _connectionError;
            set => SetProperty(ref _connectionError, value);
        }

        public bool IsComplete =>
            !string.IsNullOrWhiteSpace(ServerName)
            && !string.IsNullOrWhiteSpace(DatabaseName)
            && !string.IsNullOrWhiteSpace(Label);

        public string BuildConnectionString()
        {
            var builder = new SqlConnectionStringBuilder
            {
                DataSource = ServerName,
                InitialCatalog = DatabaseName,
                TrustServerCertificate = true,
            };

            if (UseWindowsAuth)
                builder.IntegratedSecurity = true;
            else
            {
                builder.UserID = SqlLogin;
                builder.Password = SqlPassword;
            }

            return builder.ConnectionString;
        }

        public ConnectionSide ToConnectionSide()
        {
            return new ConnectionSide
            {
                ServerName = ServerName,
                DatabaseName = DatabaseName,
                Label = Label,
                Tag = Tag,
            };
        }

        private async void ConnectAsync()
        {
            IsConnecting = true;
            ConnectionError = string.Empty;
            AvailableDatabases.Clear();

            try
            {
                var connStr = new SqlConnectionStringBuilder
                {
                    DataSource = ServerName,
                    TrustServerCertificate = true,
                };

                if (UseWindowsAuth)
                    connStr.IntegratedSecurity = true;
                else
                {
                    connStr.UserID = SqlLogin;
                    connStr.Password = SqlPassword;
                }

                var databases = await Task.Run(() =>
                {
                    var list = new System.Collections.Generic.List<string>();
                    using (var conn = new SqlConnection(connStr.ConnectionString))
                    {
                        conn.Open();
                        using (var cmd = conn.CreateCommand())
                        {
                            cmd.CommandText = "SELECT name FROM sys.databases WHERE state_desc = 'ONLINE' ORDER BY name";
                            using (var reader = cmd.ExecuteReader())
                            {
                                while (reader.Read())
                                    list.Add(reader.GetString(0));
                            }
                        }
                    }
                    return list;
                });

                foreach (var db in databases)
                    AvailableDatabases.Add(db);

                if (databases.Count > 0 && string.IsNullOrWhiteSpace(DatabaseName))
                    DatabaseName = databases[0];
            }
            catch (Exception ex)
            {
                ConnectionError = ex.Message;
            }
            finally
            {
                IsConnecting = false;
            }
        }
    }
}
```

## Task 4: TableDiffTreeBuilder — build structured tree from two TableModels

**Files:**
- Create: `src/SQLParity.Vsix/ViewModels/TableDiffTreeBuilder.cs`

This takes two `TableModel` instances (SideA and SideB, either may be null for new/dropped) and builds a tree of view-model items for the detail view. Structure:

```
Table: [dbo].[Orders]
├── Columns
│   ├── 🟢+ OrderId (INT, NOT NULL, IDENTITY)
│   ├── 🟡~ Name (NVARCHAR(100) → NVARCHAR(200))
│   └── 🔴− OldColumn (will be dropped)
├── Indexes
│   ├── PK_Orders (clustered, primary key)
│   └── IX_Orders_Date (nonclustered)
├── Foreign Keys
│   └── FK_Orders_Categories
├── Check Constraints
│   └── CK_Orders_Total
└── Triggers
    └── TR_Orders_Audit
```

Each node shows: status icon, risk icon (for changed items), name, and a summary of key properties.

- [ ] **Step 1: Create TableDiffTreeBuilder**

Create `src/SQLParity.Vsix/ViewModels/TableDiffTreeBuilder.cs`:

```csharp
using System;
using System.Collections.Generic;
using System.Collections.ObjectModel;
using System.Linq;
using SQLParity.Core.Model;

namespace SQLParity.Vsix.ViewModels
{
    /// <summary>
    /// Builds a structured tree from two TableModels for the table detail view.
    /// Shows columns, indexes, foreign keys, check constraints, and triggers
    /// with diff highlighting.
    /// </summary>
    public static class TableDiffTreeBuilder
    {
        public static ObservableCollection<TableTreeNode> Build(
            TableModel tableA, TableModel tableB,
            IReadOnlyList<ColumnChange> columnChanges)
        {
            var root = new ObservableCollection<TableTreeNode>();

            // Columns section
            var columnsGroup = new TableTreeNode { DisplayName = "Columns", IsGroup = true };
            BuildColumnNodes(columnsGroup, tableA, tableB, columnChanges);
            root.Add(columnsGroup);

            // Indexes section
            var indexesGroup = new TableTreeNode { DisplayName = "Indexes", IsGroup = true };
            BuildChildNodes(indexesGroup,
                tableA?.Indexes, tableB?.Indexes,
                idx => idx.Name, idx => FormatIndex(idx));
            root.Add(indexesGroup);

            // Foreign Keys section
            var fksGroup = new TableTreeNode { DisplayName = "Foreign Keys", IsGroup = true };
            BuildChildNodes(fksGroup,
                tableA?.ForeignKeys, tableB?.ForeignKeys,
                fk => fk.Name, fk => $"→ [{fk.ReferencedTableSchema}].[{fk.ReferencedTableName}]");
            root.Add(fksGroup);

            // Check Constraints section
            var checksGroup = new TableTreeNode { DisplayName = "Check Constraints", IsGroup = true };
            BuildChildNodes(checksGroup,
                tableA?.CheckConstraints, tableB?.CheckConstraints,
                chk => chk.Name, chk => chk.Definition);
            root.Add(checksGroup);

            // Triggers section
            var triggersGroup = new TableTreeNode { DisplayName = "Triggers", IsGroup = true };
            BuildChildNodes(triggersGroup,
                tableA?.Triggers, tableB?.Triggers,
                trig => trig.Name, trig => FormatTrigger(trig));
            root.Add(triggersGroup);

            return root;
        }

        private static void BuildColumnNodes(
            TableTreeNode group,
            TableModel tableA, TableModel tableB,
            IReadOnlyList<ColumnChange> columnChanges)
        {
            // Use columns from both sides, merged
            var colsA = tableA?.Columns ?? (IReadOnlyList<ColumnModel>)Array.Empty<ColumnModel>();
            var colsB = tableB?.Columns ?? (IReadOnlyList<ColumnModel>)Array.Empty<ColumnModel>();

            var changeByName = new Dictionary<string, ColumnChange>(StringComparer.OrdinalIgnoreCase);
            if (columnChanges != null)
            {
                foreach (var cc in columnChanges)
                    changeByName[cc.ColumnName] = cc;
            }

            // All columns from SideA
            foreach (var col in colsA)
            {
                if (changeByName.TryGetValue(col.Name, out var cc))
                {
                    group.Children.Add(new TableTreeNode
                    {
                        DisplayName = FormatColumnWithChange(col, cc),
                        Status = cc.Status,
                        Risk = cc.Risk,
                        StatusIcon = ChangeTreeItemViewModel.GetStatusIcon(cc.Status),
                        RiskIcon = ChangeTreeItemViewModel.GetRiskIcon(cc.Risk),
                        IsStrikethrough = cc.Status == ChangeStatus.Dropped,
                    });
                }
                else
                {
                    group.Children.Add(new TableTreeNode
                    {
                        DisplayName = FormatColumn(col),
                    });
                }
            }

            // Columns only in SideB (dropped)
            foreach (var col in colsB)
            {
                if (!colsA.Any(c => string.Equals(c.Name, col.Name, StringComparison.OrdinalIgnoreCase)))
                {
                    if (changeByName.TryGetValue(col.Name, out var cc))
                    {
                        group.Children.Add(new TableTreeNode
                        {
                            DisplayName = FormatColumn(col),
                            Status = ChangeStatus.Dropped,
                            Risk = cc.Risk,
                            StatusIcon = "−",
                            RiskIcon = ChangeTreeItemViewModel.GetRiskIcon(cc.Risk),
                            IsStrikethrough = true,
                        });
                    }
                }
            }
        }

        private static void BuildChildNodes<T>(
            TableTreeNode group,
            IReadOnlyList<T> listA, IReadOnlyList<T> listB,
            Func<T, string> getName, Func<T, string> getDetail)
        {
            var itemsA = listA ?? (IReadOnlyList<T>)Array.Empty<T>();
            var itemsB = listB ?? (IReadOnlyList<T>)Array.Empty<T>();

            var namesA = new HashSet<string>(itemsA.Select(getName), StringComparer.OrdinalIgnoreCase);
            var namesB = new HashSet<string>(itemsB.Select(getName), StringComparer.OrdinalIgnoreCase);

            // In both or only in A
            foreach (var item in itemsA)
            {
                var name = getName(item);
                var detail = getDetail(item);
                var inB = namesB.Contains(name);

                group.Children.Add(new TableTreeNode
                {
                    DisplayName = $"{name}  {detail}",
                    StatusIcon = inB ? "" : "+",
                    Status = inB ? (ChangeStatus?)null : ChangeStatus.New,
                });
            }

            // Only in B (dropped)
            foreach (var item in itemsB)
            {
                var name = getName(item);
                if (!namesA.Contains(name))
                {
                    group.Children.Add(new TableTreeNode
                    {
                        DisplayName = $"{name}  {getDetail(item)}",
                        StatusIcon = "−",
                        RiskIcon = "🔴",
                        Status = ChangeStatus.Dropped,
                        IsStrikethrough = true,
                    });
                }
            }
        }

        private static string FormatColumn(ColumnModel col)
        {
            var type = col.DataType;
            if (col.MaxLength > 0)
                type += $"({col.MaxLength})";

            var parts = new List<string> { col.Name, type };
            if (!col.IsNullable) parts.Add("NOT NULL");
            else parts.Add("NULL");
            if (col.IsIdentity) parts.Add("IDENTITY");
            if (col.DefaultConstraint != null)
                parts.Add($"DEFAULT {col.DefaultConstraint.Definition}");

            return string.Join("  ", parts);
        }

        private static string FormatColumnWithChange(ColumnModel col, ColumnChange cc)
        {
            if (cc.Status == ChangeStatus.New)
                return $"{FormatColumn(col)}";

            if (cc.Status == ChangeStatus.Modified && cc.SideB != null)
            {
                // Show what changed
                var changes = new List<string>();
                if (!string.Equals(col.DataType, cc.SideB.DataType, StringComparison.OrdinalIgnoreCase)
                    || col.MaxLength != cc.SideB.MaxLength)
                {
                    var oldType = cc.SideB.DataType + (cc.SideB.MaxLength > 0 ? $"({cc.SideB.MaxLength})" : "");
                    var newType = col.DataType + (col.MaxLength > 0 ? $"({col.MaxLength})" : "");
                    changes.Add($"{oldType} → {newType}");
                }
                if (col.IsNullable != cc.SideB.IsNullable)
                    changes.Add(col.IsNullable ? "→ NULL" : "→ NOT NULL");

                return $"{col.Name}  {string.Join(", ", changes)}";
            }

            return FormatColumn(col);
        }

        private static string FormatIndex(IndexModel idx)
        {
            var parts = new List<string>();
            if (idx.IsClustered) parts.Add("clustered");
            if (idx.IsPrimaryKey) parts.Add("primary key");
            else if (idx.IsUniqueConstraint) parts.Add("unique");
            if (idx.HasFilter) parts.Add("filtered");
            return string.Join(", ", parts);
        }

        private static string FormatTrigger(TriggerModel trig)
        {
            var events = new List<string>();
            if (trig.FiresOnInsert) events.Add("INSERT");
            if (trig.FiresOnUpdate) events.Add("UPDATE");
            if (trig.FiresOnDelete) events.Add("DELETE");
            return string.Join(", ", events);
        }
    }

    /// <summary>
    /// A node in the table detail tree view.
    /// </summary>
    public class TableTreeNode : ViewModelBase
    {
        public string DisplayName { get; set; } = string.Empty;
        public string StatusIcon { get; set; } = string.Empty;
        public string RiskIcon { get; set; } = string.Empty;
        public bool IsGroup { get; set; }
        public bool IsStrikethrough { get; set; }
        public ChangeStatus? Status { get; set; }
        public RiskTier? Risk { get; set; }

        public ObservableCollection<TableTreeNode> Children { get; } = new ObservableCollection<TableTreeNode>();
    }
}
```
